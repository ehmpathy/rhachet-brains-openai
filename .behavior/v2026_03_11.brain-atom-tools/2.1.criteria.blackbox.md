# blackbox criteria: brain atom tool use

## episode experience

# usecase.1 = tool invocation flow

given(a brain atom with tools plugged)
  when(brain needs external data to answer)
    then(output is null)
      sothat(caller knows brain deferred)
    then(calls.tools contains invocations)
      sothat(caller knows which tools to execute)
    then(each invocation has exid, slug, input)
      sothat(caller can correlate and execute)

  when(caller executes tools and continues episode)
    then(prompt accepts BrainPlugToolExecution array)
      sothat(tool results feed back to brain)
    then(episode continues from prior state)
      sothat(brain has full context)

  when(brain has tool results and can answer)
    then(output is defined)
      sothat(caller receives final answer)
    then(calls is null)
      sothat(caller knows no more tools needed)

# usecase.2 = direct output (no tool use)

given(a brain atom with tools plugged)
  when(brain can answer without tools)
    then(output is defined)
      sothat(caller receives answer directly)
    then(calls is null)
      sothat(caller skips tool execution loop)

# usecase.3 = no tools plugged

given(a brain atom without tools plugged)
  when(ask is called)
    then(output is always defined)
      sothat(extant behavior preserved)
    then(calls is always null)
      sothat(type system enforces no null checks)

# usecase.4 = multiple tool invocations

given(a brain atom with multiple tools plugged)
  when(brain needs data from multiple tools)
    then(calls.tools contains all invocations)
      sothat(caller can execute all in parallel)
    then(each invocation references distinct tool slug)
      sothat(caller can dispatch to correct tool)

  when(caller continues with multiple tool results)
    then(prompt accepts array of BrainPlugToolExecution)
      sothat(all results feed back together)

# usecase.5 = multi-turn tool loop

given(a brain atom with tools plugged)
  when(brain needs multiple rounds of tool use)
    then(each round returns calls until done)
      sothat(caller loops until output defined)
    then(episode accumulates context across rounds)
      sothat(brain reasons over full history)

  when(final round completes)
    then(output is defined)
    then(calls is null)
      sothat(loop terminates)

## exchange experience

# exchange.1 = ask with string prompt

given(tools are plugged)
  when(prompt is string)
    then(brain processes as initial or follow-up question)
    then(returns output OR calls, not both)

# exchange.2 = ask with tool execution results

given(tools are plugged)
  when(prompt is BrainPlugToolExecution array)
    then(brain processes as tool result continuation)
    then(episode must be provided via on.episode)

# exchange.3 = tool execution signals

given(tool execution completed)
  when(signal is 'success')
    then(output contains tool result)
  when(signal is 'error:constraint')
    then(output.error describes input/validation failure)
  when(signal is 'error:malfunction')
    then(output.error describes system failure)

# exchange.4 = output schema without tools

given(no tools plugged)
  when(output schema provided)
    then(brain returns structured output per schema)
      sothat(extant behavior preserved)

# exchange.5 = tools without output schema

given(tools plugged)
  when(no output schema provided OR output schema provided)
    then(brain can still invoke tools)
      sothat(tool use works regardless of output schema)

## boundary conditions

# boundary.1 = tool slug validation

given(brain returns tool invocation)
  when(invocation.slug matches plugged tool)
    then(invocation is valid)
  when(invocation.slug does not match any plugged tool)
    then(error is surfaced)

# boundary.2 = episode required for continuation

given(continuing with tool results)
  when(on.episode is provided)
    then(continuation succeeds)
  when(on.episode is missing)
    then(error is surfaced)

# boundary.3 = empty tools array

given(plugs.tools is empty array)
  when(ask is called)
    then(behaves as if no tools plugged)
    then(output always defined, calls always null)
