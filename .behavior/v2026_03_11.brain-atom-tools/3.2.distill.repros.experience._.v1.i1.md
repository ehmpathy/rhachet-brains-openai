## reproduction feasibility

### test utilities available

- `given/when/then/useThen` from test-fns
- `genBrainPlugToolDeclaration` from rhachet for mock tools
- `asBrainPlugToolDict` from rhachet for slug lookup
- `genArtifactGitFile` for test briefs

### setup required

1. mock tools with zod schemas
2. brain atom instance with gpt-4o-mini (fast)
3. openai api key via env

### concrete test sketch

```ts
import { genBrainPlugToolDeclaration, asBrainPlugToolDict } from 'rhachet';
import { given, when, then, useThen } from 'test-fns';
import { z } from 'zod';

describe('genBrainAtom', () => {
  given('[case1] tools are plugged and brain needs external data', () => {
    const lookupTool = genBrainPlugToolDeclaration({
      slug: 'test.lookup',
      schema: {
        input: z.object({ key: z.string() }),
        output: z.object({ value: z.string() }),
      },
      execute: async ({ invocation }) => ({
        value: `result:${invocation.input.key}`,
      }),
    });
    const tools = [lookupTool] as const;
    const toolDict = asBrainPlugToolDict(tools);

    when('[t0] ask requires tool', () => {
      const result = useThen('it defers to tool', async () =>
        genBrainAtom({ slug: 'openai/gpt/4o-mini' }).ask({
          prompt: 'lookup the value for key "test-key"',
          plugs: { tools },
        }),
      );

      then('output is null', () => {
        expect(result.output).toBeNull();
      });

      then('calls.tools is defined', () => {
        expect(result.calls?.tools).toBeDefined();
        expect(result.calls?.tools.length).toBeGreaterThan(0);
      });

      then('invocation has exid, slug, input', () => {
        const invocation = result.calls?.tools[0];
        expect(invocation?.exid).toBeDefined();
        expect(invocation?.slug).toEqual('test.lookup');
        expect(invocation?.input).toHaveProperty('key');
      });
    });

    when('[t1] tool results provided', () => {
      // first call to get invocation
      const firstResult = useThen('first ask defers', async () =>
        genBrainAtom({ slug: 'openai/gpt/4o-mini' }).ask({
          prompt: 'lookup the value for key "test-key"',
          plugs: { tools },
        }),
      );

      // execute tool and continue
      const finalResult = useThen('continuation completes', async () => {
        const invocation = firstResult.calls?.tools[0];
        const execution = await toolDict[invocation.slug].execute(
          { invocation },
          { log: console },
        );
        return genBrainAtom({ slug: 'openai/gpt/4o-mini' }).ask({
          on: { episode: firstResult.episode },
          prompt: [execution],
          plugs: { tools },
        });
      });

      then('output is defined', () => {
        expect(finalResult.output).toBeDefined();
      });

      then('calls is null', () => {
        expect(finalResult.calls).toBeNull();
      });
    });
  });
});
```
